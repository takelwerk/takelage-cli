def git_check_bit
  log.debug 'Check if we are on the git bit branch'

  return false unless git_check_workspace

  branch = _git_check_bit_get_branch
  log.debug "We are on git branch \"#{branch}\""

  branch == config.active['git_bit_branch']
end
