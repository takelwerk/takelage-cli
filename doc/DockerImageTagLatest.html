def docker_image_tag_latest
  log.debug 'Getting latest docker image tag'

  return false unless docker_check_running

  tags = docker_image_tag_list

  tag_latest = tags[-1]

  log.debug "Latest docker tag: #{tag_latest}"

  tag_latest
end
